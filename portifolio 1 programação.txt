Prática 1

algoritmo "CalculaArea"
var
   base, altura, area: real
inicio
   escreval("Digite a largura: ")
   leia(base)
   escreval("Digite a altura: ")
   leia(altura)
   
   area <- base * altura
   
   escreva("A área total do retângulo é de: ", area)
fimalgoritmo

Prática 2

algoritmo "MediaEscolar"
var
   nome: caractere
   n1, n2, n3, media: real
inicio
   escreva("Nome do aluno: ")
   leia(nome)
   escreva("Digite a primeira nota: ")
   leia(n1)
   escreva("Digite a segunda nota: ")
   leia(n2)
   escreva("Digite a terceira nota: ")
   leia(n3)
   
   media <- (n1 + n2 + n3) / 3
   
   escreva("O aluno ", nome, " obteve a média de: ", media:2:2)
fimalgoritmo
